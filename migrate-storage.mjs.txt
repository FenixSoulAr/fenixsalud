// ─── HELPERS ──────────────────────────────────────────────────────
function ask(question) {
  const rl = createInterface({ input: process.stdin, output: process.stdout });
  return new Promise(resolve => rl.question(question, ans => { rl.close(); resolve(ans); }));
}

function askPassword(question) {
  const rl = createInterface({ input: process.stdin, output: process.stdout });
  return new Promise(resolve => {
    process.stdout.write(question);
    process.stdin.setRawMode?.(true);
    let pwd = '';
    process.stdin.resume();
    process.stdin.setEncoding('utf8');
    const onData = (ch) => {
      if (ch === '\n' || ch === '\r' || ch === '\u0003') {
        process.stdin.setRawMode?.(false);
        process.stdin.pause();
        process.stdin.removeListener('data', onData);
        process.stdout.write('\n');
        resolve(pwd);
      } else if (ch === '\u007f') {
        pwd = pwd.slice(0, -1);
      } else {
        pwd += ch;
        process.stdout.write('*');
      }
    };
    process.stdin.on('data', onData);
  });
}

async function apiPost(url, headers, body) {
  const res = await fetch(url, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json', ...headers },
    body: body instanceof Buffer ? body : JSON.stringify(body),
  });
  return { status: res.status, data: res.headers.get('content-type')?.includes('json') ? await res.json() : await res.text() };
}

      // Descargar archivo
      const dlRes = await fetch(downloadUrl);
      if (!dlRes.ok) throw new Error(`Download HTTP ${dlRes.status}`);
      const fileData = Buffer.from(await dlRes.arrayBuffer());

      // Subir al nuevo proyecto
      const upRes = await fetch(`${NEW_URL}/storage/v1/object/${BUCKET}/${path}`, {
        method: 'POST',
        headers: {
          apikey: NEW_SR,
          Authorization: `Bearer ${NEW_SR}`,
          'Content-Type': mime,
          'x-upsert': 'true',
        },
        body: fileData,
      });
      if (!upRes.ok) {
        const txt = await upRes.text();
        throw new Error(`Upload HTTP ${upRes.status}: ${txt.substring(0, 80)}`);
      }

      console.log(`✅  (${(fileData.length / 1024).toFixed(0)} KB)`);
    } catch (e) {
      console.log(`❌  ${e.message.substring(0, 60)}`);
      failed.push({ path, error: e.message });
    }
  }

  // 5. Resumen
  const ok = FILES.length - failed.length;
  console.log('\n' + '─'.repeat(60));
  if (ok === FILES.length) {
    console.log(`🎉 Migración completa: ${ok}/${FILES.length} archivos OK`);
  } else {
    console.log(`⚠️  ${ok}/${FILES.length} archivos OK  |  ${failed.length} errores:`);

    console.log('\n  → Volvé a ejecutar el script para reintentar los errores.');
  }
  console.log('─'.repeat(60) + '\n');
}

main().catch(e => {

});
